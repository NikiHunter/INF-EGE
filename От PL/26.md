f = open('26-112.txt')
n, m = map(int, f.readline().split())
atms = [0]*n
atmstat = [0]*n
clients = []
lasttm = 0
for s in f:
    st, tm = map(int, s.split())
    clients.append([st, tm])
clients.sort(key = lambda x: x[0])
for s in range(len(clients)):
    st = clients[s][0]
    tm = clients[s][1]
    mint = 10**8
    minatm = 0
    for i in range(n):
        if st >= atms[i] and st <= 1440:
            atms[i] = st+tm
            atmstat[i]+=1
            lasttm = st
            break
        else:
            if atms[i] < mint:
                mint = atms[i]
                minatm = i
    else:
        if atms[minatm] <= 1440:
            lasttm = atms[minatm]
            atms[minatm] += tm
            atmstat[minatm] +=1
print(max(atmstat), lasttm)

f = open('26-125.txt')
d, p = map(int, f.readline().split())

pots = [-1]*p
gn = []
kz = 0
maxz = 0
for s in f:
    st, mn = map(int, s.split())
    gn.append([st, mn])

gn.sort()

for s in range(d):
    st = gn[s][0]
    tm = gn[s][1]//2

    for i in range(p):
        if pots[i] <= st and st < 1440:
            if st + tm + 2 > 1440:
                kz = 1440 - 2 - st
                maxz = max(maxz, 1440 - 2 - st)
            else:
                kz += tm
                maxz = max(maxz, tm)
            pots[i] = st + tm + 2
            break
    else:
        minp = min(pots)
        for i in range(p):
            if pots[i] == minp:
                if pots[i] < 1440:
                    if pots[i] + tm + 2 > 1440:
                        kz = 1440 - 2 - pots[i]
                        maxz = max(maxz, 1440 - 2 - pots[i])
                    else:
                        kz += tm
                        maxz = max(maxz, tm)
                    pots[i] += tm + 2
print(kz, maxz)

f = open('26-122.txt')

    st, end = map(int, s.split())
    cl.append([st, end])
cl.sort()
for s in cl:
    st = s[0]
    end = s[1]
    fl = 0
    for i in range(len(houses)):
        for j in range(kd):
            if houses[i][j] < st:
                houses[i][j] = end
                fl = 1
                lastt = st
                break
        if fl == 1:
            break
    else:
        houses.append([0]*kd)
        lines+=1
        for j in range(kd):
                if houses[lines-1][j] < st:
                    houses[lines-1][j] = end
                    fl = 1
                    lastt = st
                    break
kz = 0
for i in range(len(houses)):
    for j in range(kd):
        if houses[i][j] > lastt:
            kz+=1

print(lines, kz)
