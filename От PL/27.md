    return ((x1 - x2)**2 + (y1 - y2)**2)**0.5

    minr = 10**10

def dist(x1, x2, y1, y2): return ((x1-x2)**2 + (y1-y2)**2)**0.5
def fa(a):
    mind = 10**8
    a1, a2 = 0, 0
    for p in range(len(a)):
        for j in range(len(a)):
            if a[p]!=a[j]:
                s = dist(a[p][0], a[j][0], a[p][1], a[j][1])
                if s < mind:
                    mind = s
                    a1 = a[p]
                    a2 = a[j]
    return [a1[0] + a2[0], a1[1] + a2[1]]

f = open('27A 4.txt')

from math import *
from turtle import *
from random import *

def visual(allcl):
    tracer(0)
    screensize(1500, 1500)
    up()
    r = 100
    for cl in allcl:
        color = (random(), random(), random())
        for p in cl:
            goto(p[0]*r, p[1]*r)

def cent(a):
    mins = 10**8
    for i in a:
        s = sum([dist(i, p1) for p1 in a])
        if s < mins:
            mins = s
            res = i
    return res

a = [[float(x) for x in s.replace(',', '.').split()] for s in f]

allcl = []
E = 0.2294
while len(a) > 0:
    curcl = [a[-1]]
    a.pop()
    for p in curcl:
        a1 = a.copy()
        for x in a1:
            if dist(x, p) < E:
                curcl.append(x)
                a.remove(x)
    allcl.append(curcl)

allx, ally = 0, 0

from math import *
from turtle import *
from random import *

def visual(allcl):
    tracer(0)
    screensize(1500, 1500)
    up()
    r = 20
    for cl in allcl:
        color = (random(), random(), random())
        for p in cl:
            goto(p[0]*r, p[1]*r)
            dot(7, color)

def cent(a):
    mins = 10**8
    for i in a:
        s = sum([dist(i, p1) for p1 in a])
        if s < mins:
            mins = s
            res = i
    return res

f = open('27B.txt')
a1 = [[float(x) for x in s.replace(',', '.').split()] for s in f]

for e in range(1, 20):
    a = a1.copy()
    allcl = []
    E = e/10
    while len(a) > 0:
        curcl = [a[-1]]
        a.pop()
        for p in curcl:
            for x in a[:]:
                if dist(x, p) < E:
                    curcl.append(x)
                    a.remove(x)
        allcl.append(curcl)
    if len(allcl) == 3:
        resx, resy = 0, 0
        for cl in allcl:
            x,y = cent(cl)
            resx+=x
            resy+=y
        print(int(resx/len(allcl)*10000), int(resy/len(allcl)*10000))
        visual(allcl)
        break
