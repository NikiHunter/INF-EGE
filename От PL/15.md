1. ОТРЕЗКИ
```py
from itertools import combinations


def f(x: int) -> bool:
    P = 1 <= x <= 39
    Q = 23 <= x <= 58
    A = a1 <= x <= a2

    return (P <= (not Q)) <= (not A)


Ox = [i / 4 for i in range(0, 59 * 4)]
m = []

for a1, a2 in combinations(Ox, 2):
    if all(f(x) == 1 for x in Ox):
        m.append(a2 - a1)
print(min(m))
```
2. ЧИСЛОВЫЕ ВЫРАЖЕНИЯ
```py
def f(x: int, y: int) -> bool:
    return (2 * x + y != 70) or (x < y) or (a < x)


for a in range(1000):
    if all(f(x, y) == 1 for x in range(100) for y in range(100)):
        print(a)
```
3. КОНЪЮНКЦИЯ
