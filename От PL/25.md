def prime(n):
    return n > 1 and all(n%i!=0 for i in range(2, int(n**0.5) + 1))

k = 0
n = 1_000_001
while k < 5:
    d = []
    for i in range(2, int(n**0.5) + 1):
        if n%i==0:
            if prime(i):
                d.append(i)
            if i != n//i and prime(n//i):
                d.append(n//i)
    if len(d) > 0:
        s = max(d) + min(d)
        if s > 1000 and s%10==5:
            k+=1
            print(n, s)

def prime(n):
    return n > 1 and all(n%i!=0 for i in range(2, int(n**0.5) + 1))

def findD(n):
    d = []
    for i in range(1, int(n**0.5) + 1):
        if n%i==0:
            if prime(i): d.append(i)
            if i!=n//i and prime(n//i): d.append(n//i)
